class Solution {
    public int lengthOfLongestSubstring(String s) {
        HashSet<Character> h1 = new HashSet<>();
        int maxlen = 0;
        int j = 0;
        for (int i = 0; i < s.length(); i++) {
            while (h1.contains(s.charAt(i))) {
                h1.remove(s.charAt(j));
                j++;
            }
            h1.add(s.charAt(i));
            maxlen = Math.max(i - j + 1, maxlen);
        }
        return maxlen;
    }
}

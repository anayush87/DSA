 [1004. Max Consecutive Ones III Solution](https://leetcode.com/submissions/detail/1770078620/)

 #Code

class Solution {
    public int longestOnes(int[] nums, int k) {
        int count=0;
        int n=nums.length;
        int i=0;
        int maxlen=0;
        for(int j=0;j<n;j++)
        {
            if(nums[j]==0)
            {
                count++;
            }
            while(count>k)
            {
                if (nums[i] == 0) {
                    count--;
                }
                i++;
            }
            maxlen = Math.max(maxlen, j - i + 1);
        }
        return maxlen;
    }
}
